Q9: Traffic Signal Program using switch case:

"red" → Stop
"yellow" → Ready
"green" → Go
Default → Invalid Signal

let signal = "green";

switch (signal) {
  case "red":
    console.log("Stop");
    break;

  case "yellow":
    console.log("Ready");
    break;

  case "green":
    console.log("Go");
    break;

  default:
    console.log("Invalid Signal");
}

// Q11: Print numbers from 1 to 20 using for loop.

// Q12: Print only odd numbers from 1 to 20
console.log(" Print odd numbers 1 to 20");

for (let i = 1; i <= 20; i++) {
  if (i % 2 !== 0) {
    console.log(i);
  }
}
console.log("\n");


// Q13: Using while loop print numbers from 10 to 1
console.log(" Print numbers 10 to 1 using while");


let i = 10;

while (i >= 1) {
  console.log(i);
  i--;
}
console.log("\n");



// Q14: Using do-while loop print numbers from 1 to 5
console.log(" Print numbers 1 to 5 using do-while");


let j = 1;

do {
  console.log(j);
  j++;
} while (j <= 5);

console.log("\n");
