 function calculate Salary(basic Salary, bonus Percentage) {
     
    let bonus = (basic Salary * bonus Percentage) / 100;

    console.log("After Bonus Salary:"+ salary After Bonus);

    console.log("Final Salary: " + final Salary);

    return final Salary;
}

2. Student Result System:

Create generate Result(name, marks Array)
- Calculate total
- Calculate average
- Decide Grade (A/B/C/Fail)
- Return result object

function generate Result(name, marks Array) {

    // calculate total
    let total = 0;
    for (let i = 0; i < marks Array. length; i++) {
        total += marks Array[i];
    }

    // calculate average
    let average = total / marks Array. Length;

    // decide grade
    let grade;
    if (average >= 90) {
        grade = "A";
    } else if (average >= 75) {
        grade = "B";
    } else if (average >= 50) {
        grade = "C";
    } else {
        grade = "Fail";
    }

    // result object
    let result = {
        student Name: name,
        total Marks: total,
        average Marks: average,
        grade: grade
    };

    console.log("---- Student Result ----");
    console.log("Name: " + result .student Name);
    console.log("Total: " + result .total Marks);
    console.log("Average: " + result .average Marks);
    console.log("Grade: " + result .grade);

    return result;
}

// function call
generate Result(" Prathyusha", [80, 70, 90, 85, 75]);

output:
  Name : Prathyusha
 total : 400
average: 80
 grade :B

6. Bank Transaction System:

Create transaction(amount, type, callback)
- If deposit → add
- If withdraw → subtract
- Call send SMS()

let balance = 9000;

function send SMS() {
    console.log("SMS: Transaction successful");
}

function transaction(amount, type, callback) {

    if(type === "deposit"){
        balance = balance + amount;
        console.log("Amount deposited:", amount);
    }
    else if(type === "withdraw"){
        balance = balance - amount;
        console.log("Amount withdrawn:", amount);
    }
    else{
        console.log("Invalid transaction type");
        return;
    }

    console.log("Current balance:", balance);

    // callback function
    callback();
}

// function call
transaction(2000, "deposit", send SMS);

// Hide config using IIFE
const App Config = (function () {
  let tax = 0.05; // private
  return {
    get Tax: function () {
      return tax;
    }
  };
})();

// Cart array
let cart = [];

// Add to cart
function add To Cart(product, price) {
  cart. Push({ product, price });
  console.log(product + " added to cart");
}

// Calculate total
function calculate Total() {
  let total = 0;
  for (let item of cart) {
    total += item. Price;
  }

  // add tax
  total += total * App Config .get Tax();
  console.log("Total with tax:", total);
  return total;
}

// Apply discount using currying
function apply Discount(discount) {
  return function (amount) {
    return amount - amount * discount;
  };
}

// Generate coupon using generator
function* generate Coupon() {
  const coupons = ["10% OFF", "20% OFF", "50% OFF"];
  for (let c of coupons) {
    yield c;
  }
}

// Process payment using callback
function process Payment(amount, callback) {
  console.log("Processing payment of:", amount);
  set Timeout(() => {
    callback("Payment Successful");
  }, 2000);
}

Concept Questions:

1. Difference between function declaration & expression?

* Function Declaration:

// - Defined with the function keyword at the start
// - Hoisted completely (can be called before declaration)

 * Function Expression:

// - Assigned to a variable
// - Not hoisted (cannot be called before assignment)


2. What is higher order function?

.A higher-order function is a function that can take another function as an argument, return a function, or both. 
.It allows for more flexible and reusable code by enabling functions to operate on other functions.


3. Real-time example of generator?

. Generators are used when values need to be produced one at a time.

 example: coupon generator in shopping website: in e-commerce websites(Amazon . Flipkart),
 coupons are generated one by one when user clicks " get coupon".
 instead of showing all coupons at once, generator gives next coupon when needed.


4. Why do we use IIFE?

// IIFE (Immediately Invoked Function Expression) is used to create a new scope and avoid polluting the global namespace.

     var                                      let                                                const
